void printPrime(int left, int right)
{
    //TODO
    for (int n = left; n <= right; n++)
    {
        int a = 0;
        for (int i = 1; i <= n; i++)
        {
            if (n % i == 0)
                a += 1;
        }
        if (a == 2)
            cout << n << " ";
    }
    if (left < right)
        cout << endl;
    if (left > right)
        cout << "Invalid range" << endl;
}
